function somaDivisores(x) {
    let soma = 1;
    for (let i = 2; i < x; i++) {
        if (x % i === 0) {
            soma += i;
        }
   }
        return soma;
}
                  
const x = 8;
console.log(somaDivisores(x)); // 7
